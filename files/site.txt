    <script>
        // Just drop your file paths here, fam!
        const games = [
            { name: "Minecraft v1.8", path: "files/minecraft.txt", thumb: "https://cdn2.steamgriddb.com/thumb/45ec4f558dca1921c3f77b0a787b204f.jpg" },
            { name: "Bad Piggies", path: "files/bp.txt", thumb: "https://static.wikia.nocookie.net/angrybirds/images/b/ba/Bad_Piggies_Cartoon.PNG" }
        ];

        const grid = document.getElementById('game-grid');
        const modal = document.getElementById('modal');
        const frame = document.getElementById('game-frame');

        async function loadGame(path) {
            try {
                const response = await fetch(path);
                const text = await response.text();
                const blob = new Blob([text], { type: 'text/html' });
                frame.src = URL.createObjectURL(blob);
                modal.classList.remove('hidden');
            } catch (e) {
                console.error("Failed to load game:", e);
                alert("Oof, couldn't grab the game file. Check your paths!");
            }
        }

        function closeGame() {
            modal.classList.add('hidden');
            frame.src = "";
        }

        games.forEach(game => {
            const card = document.createElement('div');
            card.className = "game-card glass p-4 cursor-pointer flex flex-col items-center gap-4 group";
            card.innerHTML = `
                <div class="w-full h-40 overflow-hidden rounded-xl bg-white/5 flex items-center justify-center">
                    <img src="${game.thumb}" alt="${game.name}" class="w-full h-full object-cover group-hover:scale-110 transition duration-500" onerror="this.parentElement.innerHTML='<span class=\'text-white/20\'>No Image</span>'">
                </div>
                <h2 class="text-xl font-bold tracking-tight">${game.name}</h2>
            `;
            card.onclick = () => loadGame(game.path);
            grid.appendChild(card);
        });
    </script>
